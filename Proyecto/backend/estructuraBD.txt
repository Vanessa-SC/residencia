 *** Crear base de datos; ***

 CREATE SCHEMA IF NOT EXISTS `sistemaBD` DEFAULT CHARACTER SET utf8 ;
USE `sistemaBD` ;

*** Crear Tablas; ***

CREATE TABLE IF NOT EXISTS `sistemaBD`.`USUARIO` (
  `idUSUARIO` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `nombreUsuario` VARCHAR(45) NOT NULL,
  `contrasena` VARCHAR(45) NOT NULL,
  `tipoUsuario` INT NOT NULL)
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `sistemaBD`.`CURSO` (
  `idCURSO` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `nombreCurso` VARCHAR(150) NOT NULL,
  `fechaInicio` VARCHAR(45) NOT NULL,
  `fechaFin` VARCHAR(45) NOT NULL,
  `objetivoCurso` VARCHAR(45) NOT NULL,
  `horario` VARCHAR(45) NOT NULL,
  `duracionCurso` VARCHAR(45) NOT NULL,
  `destinatarios` VARCHAR(45) NOT NULL,
  `idINSTRUCTOR` VARCHAR(45) NOT NULL)
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `sistemaBD`.`DOCUMENTO` (
  `idDOCUMENTO` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `idCURSO` INT NOT NULL,
  `descripcion` VARCHAR(45) NOT NULL,
  `valido` INT NOT NULL,
  `rutaArchivo` VARCHAR(100) NULL)
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `sistemaBD`.`CONSTANCIA` (
  `idCONSTANCIA` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `folio` VARCHAR(10) NOT NULL UNIQUE,
  `idCurso` INT NOT NULL,
  `idUSUARIO` INT NOT NULL)
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `sistemaBD`.`ASISTENCIA` (
  `idASISTENCIA` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `nombreUsuario` VARCHAR(45) NOT NULL,
  `fecha` VARCHAR(45) NOT NULL,
  `asistencia` INT NOT NULL)
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `sistemaBD`.`ENCUESTA` (
  `idENCUESTA` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `pregunta` VARCHAR(150) NOT NULL,
  `puntuacion` INT NOT NULL)
ENGINE = InnoDB;
